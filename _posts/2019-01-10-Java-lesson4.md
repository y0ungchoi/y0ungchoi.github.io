/* Array Search - find longest name*/
public String findLongestName(String [] names){
    int size = names.length;
    String longestNames = names[0];
    for(int i=0; i<size; i++){
        if(names[i].length() > longestNames.length()){
            longestNames = names[i];
        }
    }

    return longestNames;
}
